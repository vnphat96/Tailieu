			Hàm số $y$ có cực đại và cực tiểu khi phương trình $-x^2+2x-m+3=0$ có hai nghiệm phân biệt\\

		\itemch Đúng.

	\shortans[]{2024}
	\loigiai{
		Ta có $\heva{&0=\sqrt{x}\Rightarrow x=0\\&0=x-2\Rightarrow x=2\\&\sqrt{x}=x-2\Rightarrow x=4}$\\
		Dựa vào hoành độ giao điểm của ba đường ta có diện tích hình phẳng gồm hai phần. Phần thứ nhất giới hạn bởi $y=\sqrt{x},y=0$ và $x=0;x=2$. Phần thứ hai giới hạn bởi $y=\sqrt{x},y=x-2$ và $x=2;x=4$.\\
		Thể tích vật thể bằng\\
		$V=\pi\displaystyle\int _0^2\left(\sqrt{x}\right)^2\mathrm{~d}x+\pi\displaystyle\int _2^4\left|(\sqrt{x})^2-(x-2)^2\right|\mathrm{~d}x$\\
		$=\pi\displaystyle\int _0^2x\mathrm{~d}x+\pi\displaystyle\int _2^4\left[x-(x-2)^2\right]\mathrm{~d}x$
		$=\pi\cdot \dfrac{x^2}{2}\Bigg|_0^2+\pi\cdot \left(\dfrac{x^2}{2}-\dfrac{(x-2)^3}{3}\right)\Bigg|_2^4=\dfrac{16\pi}{3}$\\
		Suy ra $a=16;b=3$.\\
		Do đó $V=128\cdot 16-8\cdot 3=2024$.
	}
\end{ex}

Biết rằng xạ thủ này bắn $2$ viên đạn một cách độc lập và chỉ có một viên trúng đích, tính xác suất để đó là xạ thủ hạng $I$.
	\shortans[]{0,34}
	\loigiai{
		Gọi $A$ là biến cố \lq\lq Xạ thủ được chọn thuộc hạng $I$\rq\rq, $B$ là biến cố \lq\lq Xạ thủ được chọn thuộc hạng $II$\rq\rq, $C$ là biến cố \lq\lq Xạ thủ chỉ bắn một viên đạn trúng đích\rq\rq. Ta cần tính $\mathrm{P}(A|C)$.\\
		Ta có\\
		$\mathrm{P}(A|C)=\dfrac{\mathrm{P}(A)\cdot \mathrm{P}(C|A)}{\mathrm{P}(A)\cdot \mathrm{P}(C|A)+\mathrm{P}(B)\cdot \mathrm{P}(C|B)}=\dfrac{0{,}4\cdot 2\cdot 0{,}8\cdot 0{,}2}{0{,}4\cdot 2\cdot 0{,}8\cdot 0{,}2+0{,}6\cdot 2\cdot 0{,}3\cdot 0{,}7}=\dfrac{32}{95}\approx0{,}34$
	}
