			&=&120\cdot\left(1{,}12^{n-1}-1\right) \text{(triệu đồng)}.
		\end{eqnarray*}

		$$T\geq 340\Leftrightarrow 120\cdot\left(1{,}12^{n-1}-1\right)\geq 340\Leftrightarrow n-1\geq \log _{1{,}12}\left(\dfrac{23}{6}\right)\Leftrightarrow n\geq 12{,}8.$$

	}
