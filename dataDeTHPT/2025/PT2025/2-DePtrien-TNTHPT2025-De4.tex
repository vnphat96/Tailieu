	{\True $g(t)=kt+C$ với $C$ là hằng số}

		\begin{itemchoice}
			\itemch  Ta có $y(t) = \mathrm{e}^{g(t)} \Rightarrow y'(t) = g'(t)\cdot \mathrm{e}^{g(t)} = g'(t) \cdot y(t)$.\\

			\itemch  Ta có
			\[
			y(16) = A \cdot \mathrm{e}^{16k} = 5\cdot \mathrm{e}^{-4k} \cdot \mathrm{e}^{16k} = 5\cdot \mathrm{e}^{12k} = 5 \cdot \mathrm{e}^{12\cdot \frac{\ln 4}{6}} = 5 \cdot \mathrm{e}^{2\ln 4} = 5 \cdot 4^2 = 5\cdot 16 = 80.

			\itemch Ta có $\cos\left(\overrightarrow{u},\overrightarrow{i}\right)=\dfrac{\overrightarrow{u}\cdot\overrightarrow{i}}{\left|\overrightarrow{u}\right|\cdot\left|\overrightarrow{i}\right|} \Leftrightarrow \cos 120^\circ=\dfrac{a}{2} \Leftrightarrow a=2\cos 120^\circ=-1$.
