\begin{ex}%[2D2B4-3]
	[Chuyên Vinh 2017]
	Cho là các số $\alpha$, $\beta $ là các số thực. Đồ thị các hàm số $ y=x^{\alpha}$, $y=x^{\beta}$ trên khoảng $\left(0;+\infty\right)$ được cho trong hình vẽ bên. Khẳng định nào sau đây là đúng?
	\begin{center}
		\begin{tikzpicture}[scale=0.7, font=\footnotesize, line join=round, line cap=round,>=stealth]
			\def\xmax{6} \def\ymax{5}			
			\draw[->] (-0.9,0)--(\xmax,0) node [below]{$x$};
			\draw[->] (0,-0.9)--(0,\ymax) node [left]{$y$};
			\node at (0,0) [below left]{$O$};
			\clip (-0.9,-0.9) rectangle (\xmax-0.5,\ymax-0.2);
			\draw[smooth,samples=300,domain=0:\xmax] plot(\x,{(\x)^2.5});			
			\draw[smooth,samples=300,domain=0:\xmax] plot(\x,{(\x)^0.3});
			\node at (2.2,4.5)[right]{$y=x^{\alpha}$};
			\node at (5,1){$y=x^{\beta}$};
		\end{tikzpicture}
	\end{center}
	\choice
	{$0<\alpha<1<\beta$}
	{$\beta <0<1<\alpha$}
	{\True $0<\beta<1<\alpha$}
	{$\alpha <0<1<\beta$}
	\loigiai{
		Với $x_0>1$ ta có $x_0^{\alpha}>1\Rightarrow\alpha >0;x_0^{\beta}>1\Rightarrow\beta>0$.\\
		$x_0^{\alpha}>x_0^{\beta}\Rightarrow\alpha>\beta$.
	}

\begin{ex}%[2D2B4-3]
	[THPT – THD Nam Định-2017]
	Cho hàm số $ y=x^{-\sqrt{2}}$. Mệnh đề nào sau đây là \textbf{sai}?
	\choice
	{Hàm số có tập xác định là $\left(0;+\infty\right)$}
	{\True Đồ thị hàm số không có tiệm cận}
	{Hàm số nghịch biến trên khoảng $\left(0;+\infty\right)$}
	{Đồ thị hàm số không cắt trục hoành}
	\loigiai{      
		Tập xác định $\mathscr{D}=\left(0;+\infty\right)$, suy ra \lq\lq Hàm số nghịch biến trên khoảng $\left(0;+\infty\right)$\rq\rq\, đúng.\\
		Do $ x>0$ nên $x^{-\sqrt{2}}>0$, suy ra \lq\lq Hàm số có tập xác định là $\left(0;+\infty\right)$\rq\rq\, đúng.\\
		Ta có $y'=-\sqrt{2}.x^{-\sqrt{2}-1}<0;\forall x>0$, suy ra \lq\lq Đồ thị hàm số không có tiệm cận\rq\rq\, đúng.\\
		Ta có $\underset{x\to{0^+}}{\lim}x^{-\sqrt{2}}=+\infty$ nên đồ thị hàm số nhận $Oy$ làm tiệm cận đứng, đáp án \lq\lq Đồ thị hàm số không cắt trục hoành\rq\rq\, đúng.
	}
\end{ex}
