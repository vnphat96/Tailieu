\begin{ex}%[2D1K4-1]%[Đề minh hoạ 2020 lần 1] 
	Tổng số tiệm cận đứng và tiệm cận ngang của đồ thị hàm số $ y=\dfrac{5x^2-4x-1}{x^2-1} $ là 
	\choice
	{$ 0 $}
	{$ 1 $}
	{\True $ 2 $}
	{$ 3 $}
	\loigiai
	{
		$ \bullet $ Tiệm cận ngang:
		\[\lim \limits_{x \to \pm\infty} y = \lim \limits_{x \to \pm \infty} \dfrac{5x^2-4x-1}{x^2-1} = \lim \limits_{x \to \pm \infty} \dfrac{5-\dfrac{4}{x}-\dfrac{1}{x^2}}{1-\dfrac{1}{x^2}}=5.\]
		Suy ra đồ thị hàm số có một tiệm cận ngang $ y=5 $.\\
		$ \bullet $ Tiệm cận đứng:
		Cho $x^2=1 \Leftrightarrow \hoac{&x=1\\&x=-1}$
		\[\lim \limits_{x \to 1} y=\lim \limits_{x \to 1} \dfrac{5x^2-4x-1}{x^2-1} = \lim \limits_{x \to 1} \dfrac{(5x+1)(x-1)}{(x+1)(x-1)}=\lim \limits_{x \to 1} \dfrac{5x+1}{x+1}=3.\]
		Suy ra đường thẳng $ x=1 $ không phải là tiệm cận đứng của đồ thị hàm số.
		\[\lim \limits_{x \to -1^{+}} y=\lim \limits_{x \to -1^{+}} \dfrac{5x^2-4x-1}{x^2-1} = \lim \limits_{x \to -1^{+}} \dfrac{1}{x+1} \cdot \dfrac{5x^2-4x-1}{x-1}=-\infty \]
		Vì $\lim \limits_{x \to -1^{+}} \dfrac{1}{x+1} =+\infty$ và $\lim \limits_{x \to -1^{+}}\dfrac{5x^2-4x-1}{x-1} =-4<0$.\\
		Suy ra đồ thị hàm số có một tiệm cận đứng $ x=-1 $.\\
		Tổng cộng đồ thị hàm số có tổng số tiệm cận đứng và tiệm cận ngang là 2.
	}
\end{ex}

		\[\lim \limits_{x \to -2^{+}} \dfrac{x-2}{x^2-4}=\lim \limits_{x \to -2^{+}} \dfrac{1}{x+2}=+\infty\]
		Do đó đường thẳng $ x=-2 $ là tiệm cận đứng của đồ thị hàm số.\\

		\[\lim \limits_{x \to \pm \infty} \dfrac{x-2}{x^2-4}=\lim \limits_{x \to \pm \infty} \dfrac{1}{x+2}=0\]
		Do đó đồ thị hàm số có một đường tiệm cận ngang là đường thẳng $ y=0 $.\\

	}

	{$4$}
	\loigiai{
		Tập xác định của hàm số: $ \mathscr{D} = \left(-\infty;-\dfrac{3}{2}\right] \cup \left[0;+\infty\right)\setminus\left\{-2\right\}$.\\

		$\lim \limits_{x \rightarrow-\infty} \dfrac{\sqrt{x(4 x+6)}-2}{x+2}=\lim \limits_{x \rightarrow-\infty} \dfrac{-\sqrt{4+\dfrac{6}{x}}-\dfrac{2}{x}}{1+\dfrac{2}{x}}=-2$. \\
		$\lim \limits_{x \rightarrow-2^{\pm}} \dfrac{\sqrt{x(4 x+6)}-2}{x+2}=\lim \limits_{x \rightarrow-2^{\pm}} \dfrac{(x+2)(4 x-2)}{(x+2)(\sqrt{x(4 x+6)}+2)}=\lim \limits_{x \rightarrow-2^{\pm}} \dfrac{4 x-2}{\sqrt{x(4 x+6)}+2}=\dfrac{-5}{2}$.\\

\begin{ex}%[2D1K4-1]%[THPT Bạch Đằng Quảng Ninh 2019] . 
	Cho hàm số $y=\dfrac{x^2+3 x+2}{\sqrt{x^4-3 x^2+2}}$. Đồ thị hàm số đã cho có bao nhiêu đường tiệm cận?
	\choice
	{\True  $4$}
	{$5$}
	{$3$}
	{$6$}
	\loigiai{
		Điều kiện: $x \in(-\infty ;-\sqrt{2}) \cup(-1 ; 1) \cup(\sqrt{2} ;+\infty)$. Ta có\\
		$\lim \limits_{x \rightarrow+\infty} y=\lim \limits_{x \rightarrow-\infty} y=\lim \limits_{x \rightarrow \pm \infty} \dfrac{x^2+3 x+2}{\sqrt{x^4-3 x^2+2}}=\lim \limits_{x \rightarrow \pm \infty} \dfrac{1+\dfrac{3}{x}+\dfrac{2}{x^2}}{\sqrt{1-\dfrac{3}{x^2}+\dfrac{2}{x^4}}}=1 \Rightarrow y=1$ là đường tiệm cận ngang của đồ thị hàm số.\\
		$\lim \limits_{x \rightarrow 1^{-}} y=+\infty$ nên đường thẳng $x=1$ là đường tiệm cận đứng.\\
		\begin{eqnarray*}
			\lim \limits_{x \rightarrow(-1)^{+}} y  &=& \lim \limits_{x \rightarrow(-1)^{+}} \dfrac{(x+1)(x+2)}{\sqrt{(x+1)(x+\sqrt{2})(x-1)(x-\sqrt{2})}} \\
			&=& \lim\limits_{x \to (-1)^{+}} \dfrac{\sqrt{(x+1)}(x+2)}{\sqrt{(x+\sqrt{2})(x-1)(x-\sqrt{2})}}=0
		\end{eqnarray*}
		nên đường thẳng $x=-1$ không là đường tiệm cận đứng.\\
		$\lim \limits_{x \rightarrow(\sqrt{2})^{+}} y=+\infty$ nên đường thẳng $x=\sqrt{2}$ là đường tiệm cận đứng.\\
		$\lim \limits_{x \rightarrow(-\sqrt{2})^{-}} y=+\infty$ nên đường thẳng $x=-\sqrt{2}$ là đường tiệm cận đứng.\\
		Vậy đồ thị hàm số có $4$ đường tiệm cận (1 tiệm cận ngang, 3 tiệm cận đứng).
	}
\end{ex}

			{ x - m^2 = x - 4 } \\
			{ m^2 = 0 }
		\end{array} \Leftrightarrow \left[\begin{array} { l } 
			{ m^2 = 4 } \\
			{ m = 0 }
		\end{array} \Leftrightarrow \left[\begin{array}{l}
			m=\pm 2 \\
			m=0
		\end{array} .\right.\right.\right.
		$
		Vậy có 3 giá trị nguyên của tham số $m$ thỏa mãn yêu cầu bài toán.}
\end{ex}

	Tìm $m$ để đồ thị hàm số $y=\dfrac{2 x^2-3 x+4}{x^2+m x+1}$ có duy nhất một đường tiệm cận.
	\choice 
	{\True $m \in(-2 ; 2)$}
	{$m \in[-2 ; 2]$}
	{$m \in\{-2 ; 2\}$}
	{$m \in(2 ;+\infty)$} 
	\loigiai{
		Ta có $\lim\limits_{x \to +\infty} \dfrac{2 x^2-3 x+4}{x^2+m x+1}=2$ và $\lim\limits_{x \to -\infty} \dfrac{2 x^2-3 x+4}{x^2+m x+1}=2$ nên đồ thị hàm số có 1 tiệm cậ ngang $y=2$.\\
		Vì $2 x^2-3 x+4>0, \forall x \in \mathbb{R}$ nên đồ thị hàm số có một đường tiệm cận khi $x^2+m x+1=0$ vô nghiệm.\\
		$\Delta=m^2-4<0 \Leftrightarrow m \in(-2 ; 2)$}
