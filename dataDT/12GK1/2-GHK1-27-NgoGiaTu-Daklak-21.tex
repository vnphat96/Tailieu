			\item Hàm số $ y = \dfrac{x- 3}{x - 1} $ luôn đồng biến trên các khoảng $ (-\infty; 1)  $ và $ (1; + \infty) $.
			\item Hàm số $ y = - x^3 - 3x $ luôn có hai điểm cực trị.
			\item Hàm số $ y = x^3 + 3x $ luôn đồng biến trên $

		\end{itemize}
