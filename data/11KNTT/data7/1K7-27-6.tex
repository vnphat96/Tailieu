\subsubsection{Ví dụ mẫu}
\begin{vd}%[Nguyễn Thành Nhân]%[Dự án 11 chương trình mới]%%[1K7KQ-5]
	Cho $x$, $y$ là các số thực dương. Xét các khối chóp $S.ABC$ có $SA=x$, $BC=y$, các cạnh còn lại đều bằng $1$. Khi $x$, $y$ thay đổi, tính giá trị lớn nhất thể tích của khối chóp.
	\loigiai{
		\immini{Gọi $M$, $K$ lần lượt là trung điểm $BC$, $SA$.\\
			Khi đó $AM^2=1-\dfrac{y^2}{4}$, $MK^2=AM^2-\dfrac{x^2}{4}=1-\dfrac{x^2+y^2}{4}$.\\
			Lại có $S_{\triangle ABC}=\dfrac{1}{2}AM\cdot BC$ và $SH\cdot AM=MK\cdot SA$ \\
			từ đó suy ra $SH=\dfrac{MK\cdot SA}{AM}$.\\
			Thể tích khối chóp 
			{\allowdisplaybreaks\begin{eqnarray*}
					V_{S.ABC}&=&\dfrac{1}{3}\cdot S_{\triangle ABC}\cdot SH\\
					&=&\dfrac{1}{3}\cdot \dfrac{1}{2}AM\cdot BC\cdot \dfrac{MK\cdot SA}{AM}\\
					&=&\dfrac{1}{6}SA\cdot BC\cdot MK\\
					&=&\dfrac{1}{12}x\cdot y\cdot \sqrt{4-(x^2+y^2)}.
		\end{eqnarray*}}}
		{\begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=1]
				\path
				(0,0) coordinate (A)
				(2.4,-1.7) coordinate (B)
				(4,0) coordinate (C)
				(1.3,-0.7) coordinate (H)
				($(H)+(0,4.5)$) coordinate (S)
				($(B)!1/2!(C)$) coordinate (M)
				($(S)!1/2!(A)$) coordinate (K);
				\draw  (S)--(A)--(B)--(C)--(S)--(B);
				\draw[dashed]  (K)--(M)--(A)--(C) (S)--(H);
				\fill[black] (S) circle (1.5pt) node[above right]{$S$} (A) circle (1.5pt) node[below left]{$A$}(B) circle (1.5pt) node[below]{$B$}(C) circle (1.5pt) node[right]{$C$} (H) circle (1.5pt) node[right]{$H$} (M) circle (1.5pt) node[below right]{$M$} (K) circle (1.5pt) node[left]{$K$};
		\end{tikzpicture}}
		\noindent Theo bất đẳng thức Cô-si ta có
		\begin{eqnarray*}
			V_{S.ABC}&=&\dfrac{1}{12}x\cdot y\cdot \sqrt{4-(x^2+y^2)}\\
			&\le& \dfrac{1}{12}\sqrt{\dfrac{[x^2+y^2+(4-(x^2+y^2))]^3}{27}}=\dfrac{2\sqrt{3}}{27}.
		\end{eqnarray*}
		Vậy thế tích lớn nhất của khối chóp là $ \dfrac{2\sqrt{3}}{27} $ xảy ra khi và chỉ khi $ x=y=\dfrac{2\sqrt{3}}{3}. $}
\end{vd}

\begin{vd}%%[Nguyễn Thành Nhân]%[Dự án 11 chương trình mới]%%[1K7KQ-5]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình thoi tâm $O$ cạnh bằng $a$, $SA=SB=SC=a$. Tính giá trị lớn nhất của thể tích của khối chóp $S.ABCD$.
	\loigiai
	{
		\immini
		{
			Vì $SA=SB=SC$ nên chân đường cao $SH$ trùng với tâm đường tròn ngoại tiếp tam giác $ABC$. Suy ra $H\in BO$.\\
			Ta có $\triangle SAC = \triangle BAC$ (c-c-c) nên $OS=OB$.\\
			Mà $OB=OD$ nên $OS=OB=OD$, suy ra tam giác $SBD$ vuông tại $S$.\\
			Đặt $SD=x$. Khi đó $SH=\dfrac{ax}{\sqrt{a^2+x^2}}$.			
		}

		\allowdisplaybreaks

		Đẳng thức xảy ra khi $x^2=3a^2-x^2$ hay $SD=x=\dfrac{a\sqrt{6}}{2}$.\\
		Vậy thể tích của khối chóp $S.ABCD$ lớn nhất bằng $\dfrac{a^3}{4}$.

\begin{vd}%[Nguyễn Thành Nhân]%[Dự án 11 chương trình mới]%%[1K7KQ-5]
	Cho hình chóp $S.ABC$ có $SA=SB=SC=AB=AC=a$, $BC=2x$ với $a$ là hằng số và $x$ thay
	đổi thuộc khoảng $\left(0 ;\dfrac{a\sqrt{3}}{2}\right)$. Tính thể tích lớn nhất $V_{\max}$ của hình chóp $S.ABC$.
	\loigiai{
		\immini
		{
			Gọi $H$ là hình chiếu vuông góc của $S$ trên $(ABC)$ và $M$ là trung điểm của $BC$.\\
			Vì $SA=SB=SC$ nên $\triangle SHA=\triangle SHB=\triangle SHC$.\\
			Suy ra $HA=HB=HC$ $\Rightarrow H$ là tâm đường tròn ngoại tiếp $\triangle ABC$ và $HA=HB=HC=R$.\\
			Vì $\triangle ABC$ cân tại $A$ nên $H\in AM$.\\
			Xét $\triangle ABM\Rightarrow AM^2=AB^2-BM^2=a^2-x^2$\\
			$\Rightarrow AM=\sqrt{a^2-x^2}$.\\
		}
		{
			\begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=1]
				\path
				(0,0) coordinate (A)
				(1.7,-2) coordinate (B)
				(5,0) coordinate (C);
				\coordinate (M) at ($(B)!1/2!(C)$);
				\coordinate (H) at ($(A)!2/3!(M)$);
				\coordinate (S) at ($(H)+(0,5)$);
				\draw  (M)--(S)--(A)--(B)--(C)--(S)--(B);
				\draw[dashed]  (M)--(A)--(C) (S)--(H);
				\draw (S) circle (1.5pt) node[above right]{$S$}--(A) circle (1.5pt) node[left]{$A$}--(B) circle (1.5pt) node[below]{$B$}--(S)--(C) circle (1.5pt) node[right]{$C$};
				\draw (M) circle (1.5pt) node[below right]{$M$};
				\draw (H) circle (1.5pt) node[below]{$H$};
			\end{tikzpicture}
		}
		\noindent Diện tích tam giác $ABC$ là $S=\dfrac{1}{2}\cdot BC\cdot AM=\dfrac{1}{2}\cdot 2x\cdot \sqrt{a^2-x^2}=x\sqrt{a^2-x^2}$.\\
		Xét $\triangle ABM$ vuông tại $M\Rightarrow \sin\widehat{ABM}=\dfrac{AM}{AB}=\dfrac{\sqrt{a^2-x^2}}{a}$.\\
		Theo định lí sin trong tam giác $ABC$
		$$\dfrac{AC}{\sin\widehat{ABM}}=2R\Leftrightarrow R=\dfrac{a^2}{2\sqrt{a^2-x^2}}\Rightarrow HA=\dfrac{a^2}{2\sqrt{a^2-x^2}}.$$
		Xét $\triangle SAH$ vuông tại $H$
		$\Rightarrow SH=\sqrt{SA^2-AH^2}=\sqrt{a^2-\dfrac{a^4}{4(a^2-x^2)}}=\dfrac{a\sqrt{3a^2-4x^2}}{2\sqrt{a^2-x^2}}$.\\
		Thể tích khối chóp $S.ABC$ là
		$$V=\dfrac{1}{3}\cdot S\cdot SH=\dfrac{1}{3}\cdot x\sqrt{a^2-x^2}\cdot \dfrac{a^2\sqrt{3a^2-4x^2}}{2\sqrt{a^2-x^2}}=\dfrac{1}{6}xa\sqrt{3a^2-4x^2}=\dfrac{a}{12}\cdot 2x\cdot \sqrt{3a^2-4x^2}.$$
		Áp dụng bất đẳng thức AM-GM suy ra $2x\cdot \sqrt{3a^2-4x^2}\le \dfrac{4x^2+3a^2-4x^2}{2}=\dfrac{3a^2}{2}$.\\
		Do đó $V\le \dfrac{a^3}{8}$. Dấu bằng xảy ra khi $2x=\sqrt{3a^2-4x^2}\Leftrightarrow x=\dfrac{a\sqrt{6}}{4}$ (thỏa mãn).\\
		Vậy thể tích lớn nhất của hình chóp $S.ABC$ là $V_{\max}=\dfrac{a^3}{8}$.
	}
\end{vd}
