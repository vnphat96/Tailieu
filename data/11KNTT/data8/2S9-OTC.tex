\begin{ex}%[1D2Y1-1]

	\choice
	{ $\mathrm{P}(A)\cdot  \mathrm{P}(B)$}
	{ $\mathrm{P}(A)-\mathrm{P}(B)$}
	{ $\mathrm{P}(A)+\mathrm{P}(A \cap B)$}
	{\True  $\mathrm{P}(A)+\mathrm{P}(B)$}
	\loigiai{}
\end{ex}

\begin{ex}%[1D2Y1-1]

\begin{ex}%[1D2Y2-2]

	\choice

	{$0,48$}
	{\True $0,8$}

		Xác suất của biến cố \lq\lq  Bạn được gặp thích chơi cả hai loại\rq\rq \ là:

	}
\end{ex}
