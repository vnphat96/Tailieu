f'(x)&=&\dfrac{(3x+5)'\cdot(2x-1)-(3x+5)(2x-1)'}{(2x-1)^2}\\
&=&\dfrac{3(2x-1)-2(3x+5)}{(2x-1)^2}\\
&=&\dfrac{-13}{(2x-1)^2}.
\end{eqnarray*}
\end{enumerate}
}
\end{ex}
