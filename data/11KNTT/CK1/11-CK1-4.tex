	\end{align*}

	\begin{align*}
		N_2 = \left[10^3\cdot (1{,}01) - a\right] + \left[10^3\cdot (1{,}01) - a\right]\cdot 1\% - a = 10^3\cdot (1{,}01)^2 - \left[a\cdot (1{,}01)+a\right].
