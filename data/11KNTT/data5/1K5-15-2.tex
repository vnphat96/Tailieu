%\chapter{Giới hạn hàm số}

\begin{dang}{Phương pháp lượng liên hợp (lim hữu hạn)}
	Nếu giới hạn của dãy số ở dạng vô định thì ta sử dụng các phép biến đổi để đưa về dạng cơ bản. \\
	Một số phép biến đổi liên hợp: \\
	\begin{align*}
		f(n) - g(n) &= \dfrac{(f(n))^2 - (g(n))^2}{f(n) + g(n)} \\
		\sqrt{f(n)} - \sqrt{g(n)} &= \dfrac{f(n) - g(n)}{\sqrt{f(n)} + \sqrt{g(n)}} \\
		\sqrt{f(n)} - g(n) &= \dfrac{f(n) - (g(n))^2}{\sqrt{f(n)} + g(n)} \\
		\sqrt[3]{f(n)} - \sqrt[3]{g(n)} &= \dfrac{f(n) - g(n)}{\sqrt[3]{(f(n))^2} + \sqrt[3]{f(n)g(n)} + \sqrt[3]{(g(n))^2}}
	\end{align*}
\end{dang}

\subsubsection{Ví dụ mẫu}
\begin{vd}
    Tính giới hạn $I = \lim \left(\sqrt{n^2 - 2n + 3} - n\right)$. 
    \loigiai{
    Ta có 
    \begin{align*}
    I &= \lim \left(\sqrt{n^2 - 2n + 3} - n\right) \\
    &= \lim \dfrac{n^2 - 2n + 3 - n^2}{\sqrt{n^2 - 2n + 3} + n} \\
    &= \lim \dfrac{- 2n + 3}{\sqrt{n^2 - 2n + 3} + n} \\
    &= \lim \dfrac{- 2 + \frac{3}{n}}{\sqrt{1 - \frac{2}{n} + \frac{3}{n^2}} + 1} \\
    &= \dfrac{- 2}{\sqrt{1} + 1} = - 1
    \end{align*}
    }
\end{vd} 
\begin{vd}
    Tính giới hạn $I = \lim \left(\sqrt{n^2 + 7} - \sqrt{n^2 + 5}\right)$. 
    \loigiai{
    Ta có 
    \begin{align*}
    I &= \lim \left(\sqrt{n^2 + 7} - \sqrt{n^2 + 5}\right) \\
    &= \lim \dfrac{n^2 + 7 - (n^2 + 5)}{\sqrt{n^2 + 7} + \sqrt{n^2 + 5}} \\
    &= \lim \dfrac{2}{\sqrt{n^2 + 7} + \sqrt{n^2 + 5}} \\
    &= 0
    \end{align*}
    }
\end{vd}
\begin{vd}
    Tính giới hạn $I = \lim \left(\sqrt{n^2 + 2n} - \sqrt{n^2 - 2n}\right)$. 
    \loigiai{
    Ta có 
    \begin{align*}
    I &= \lim \left(\sqrt{n^2 + 2n} - \sqrt{n^2 - 2n}\right) \\
    &= \lim \dfrac{n^2 + 2n - (n^2 - 2n)}{\sqrt{n^2 + 2n} + \sqrt{n^2 - 2n}} \\
    &= \lim \dfrac{4n}{\sqrt{n^2 + 2n} + \sqrt{n^2 - 2n}} \\
    &= \lim \dfrac{4}{\sqrt{1 + \frac{2}{n}} + \sqrt{1 - \frac{2}{n}}} \\
    &= \dfrac{4}{\sqrt{1} + \sqrt{1}} = 2
    \end{align*}
    }
\end{vd}

\begin{vd}
    Tính giới hạn $I = \lim \left(\sqrt{2n^2 - n + 1} - \sqrt{2n^2 - 3n + 2}\right)$. 
    \loigiai{
    Ta có 
    \begin{align*}
    I &= \lim \left(\sqrt{2n^2 - n + 1} - \sqrt{2n^2 - 3n + 2}\right) \\
    &= \lim \dfrac{2n^2 - n + 1 - (2n^2 - 3n + 2)}{\sqrt{2n^2 - n + 1} + \sqrt{2n^2 - 3n + 2}} \\
    &= \lim \dfrac{2n - 1}{\sqrt{2n^2 - n + 1} + \sqrt{2n^2 - 3n + 2}} \\
    &= \lim \dfrac{2 - \frac{1}{n}}{\sqrt{2 - \frac{1}{n} + \frac{1}{n^2}} + \sqrt{2 - \frac{3}{n} + \frac{2}{n^2}}} \\
    &= \dfrac{2}{\sqrt{2} + \sqrt{2}} = \dfrac{1}{\sqrt{2}}
    \end{align*}
    }
\end{vd} 
\begin{vd}
    Tính giới hạn $I = \lim \left(n - \sqrt[3]{n^3 + 3n^2 + 1}\right)$.
    \loigiai{
    Ta có 
\begin{align*} 
I &= \lim \left(n - \sqrt[3]{n^3 + 3n^2 + 1}\right) \\
 &= \lim \dfrac{n^3 - (n^3 + 3n^2 + 1)}{n^2 + \sqrt[3]{n^3 + 3n^2 + 1} + \sqrt[3]{\left(n^3 + 3n^2 + 1\right)^2}} \\
 &= \lim \dfrac{- 3n^2 - 1}{n^2 + \sqrt[3]{n^3 + 3n^2 + 1} + \sqrt[3]{\left(n^3 + 3n^2 + 1\right)^2}} \\
 &= \lim \dfrac{- 3 - \frac{1}{n^2}}{1 + \sqrt[3]{1 + \frac{3}{n} + \frac{1}{n^3}} + \sqrt[3]{\left(1 + \frac{3}{n} + \frac{1}{n^3}\right)^2}} \\ 
 &= \dfrac{- 3}{1 + \sqrt[3]{1} + \sqrt[3]{1}} = - 1
\end{align*}
 }
\end{vd}
\subsubsection{Bài tập rèn luyện}

\begin{bt}%[1K5?E-4]%[1K5?E-4]
Tính giới hạn  $I = \lim \left(\sqrt[3]{n^3 - 2n} - n\right)$. 
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(\sqrt[3]{n^3 - 2n} - n\right) \\
    &= \lim \dfrac{n^3 - 2n - n^3}{\sqrt[3]{\left(n^3 - 2n\right)^2} + n \sqrt[3]{n^3 - 2n} + n^2} \\ 
    &= \lim \dfrac{-2n}{\sqrt[3]{\left(n^3 - 2n\right)^2} + n \sqrt[3]{n^3 - 2n} + n^2} \\
    &= \lim \dfrac{\frac{-2}{n}}{\sqrt[3]{\left(1 - \frac{2}{n^2}\right)^2} + \sqrt[3]{1 - \frac{2}{n^2}} + 1} \\
    &= 0 
    \end{align*}
    }
    \end{bt}
\begin{bt}%[1K5?E-4]%[1K5?E-4]
Tính giới hạn $I = \lim \left(\sqrt{4n^2 + 5n} - 2n\right)$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(\sqrt{4n^2 + 5n} - 2n\right) \\
    &= \lim \dfrac{4n^2 + 5n - 4n^2}{\sqrt{4n^2 + 5n} - 2n} \\ 
    &= \lim \dfrac{5n}{\sqrt{4n^2 + 5n} + 2n} \\
    &= \lim \dfrac{5}{\sqrt{4 + \frac{5}{n}} + 2} \\
    &= \dfrac{5}{\sqrt{4} + 2} = \dfrac{5}{4}
    \end{align*}
    }
    \end{bt}
 \begin{bt}%[1K5?E-4]%[1K5?E-4]
Tính giới hạn $I = \lim \left(3n - \sqrt{9n^2 + 1}\right)$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(3n - \sqrt{9n^2 + 1}\right) \\
    &= \lim \dfrac{9n^2 - (9n^2 + 1)}{3n + \sqrt{9n^2 + 1}} \\ 
    &= \lim \dfrac{- 1}{3n + \sqrt{9n^2 + 1}} \\
    &= 0
    \end{align*}
    }
    \end{bt}  
    \begin{bt}%[1K5?E-4]%[1K5?E-4]
Tính giới hạn $I = \lim \left(3n - 5 - \sqrt{9n^2 + 1}\right)$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(3n - \sqrt{9n^2 + 1}\right) \\
    &= \lim \dfrac{(3n - 5)^2 - (9n^2 + 1)}{3n - 5 + \sqrt{9n^2 + 1}} \\ 
    &= \lim \dfrac{- 30n + 24}{3n - 5 + \sqrt{9n^2 + 1}} \\
    &= \lim \dfrac{- 30 + \frac{24}{n}}{3 - \frac{5}{n} + \sqrt{\frac{1}{n^2}}} \\
    &= \dfrac{- 30}{3 + \sqrt{9}} = - 5
    \end{align*}
    }
    \end{bt}  
    \begin{bt}%[1K5?E-4]%[1K5?E-4]
        Tính giới hạn $I = \lim \left(\sqrt[3]{n + 2} - \sqrt[3]{n}\right)$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(\sqrt[3]{n + 2} - \sqrt[3]{n}\right) \\
    &= \lim \dfrac{n + 2 - n}{\sqrt[3]{(n + 2)^2} + \sqrt[3]{n(n + 2)} + \sqrt[3]{n^2}} \\ 
    &= \lim \dfrac{2}{\sqrt[3]{(n + 2)^2} + \sqrt[3]{n(n + 2)} + \sqrt[3]{n^2}} \\  
    &= 0
    \end{align*}
    }
    \end{bt} 
    \begin{bt}%[1K5?E-4]%[1K5?E-4]
        Tính giới hạn $I = \lim \dfrac{\sqrt{4n^2 + 2n} - n + 1}{\sqrt{9n^2 + n} - 2n}$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \dfrac{\sqrt{4n^2 + 2n} - n + 1}{\sqrt{9n^2 + n} - 2n} \\
    &= \lim \dfrac{\left[4n^2 + 2n - (n - 1)^2\right]\left(\sqrt{9n^2 + n} + 2n\right)}{\left(\sqrt{4n^2 + 2n} + n - 1\right)(9n^2 + n - 4n^2)} \\ 
    &= \lim \dfrac{(3n^2 + 4n - 1)\left(\sqrt{9n^2 + n} + 2n\right)}{\left(\sqrt{4n^2 + 2n} + n - 1)(5n^2 + n\right)} \\
    &= \lim \dfrac{\left(3 + \frac{4}{n} - \frac{1}{n^2}\right)\left(\sqrt{9 + \frac{1}{n}} + 2\right)}{\left(\sqrt{4 + \frac{2}{n}} + 1 - \frac{1}{n}\right)\left(5 + \frac{1}{n}\right)} \\
    &= \dfrac{3\left(\sqrt{9} + 2\right)}{5\left(\sqrt{4} + 1\right)} = 1
    \end{align*}
    }
    \end{bt}  
    \begin{bt}%[1K5?E-4]%[1K5?E-4]
        Tính giới hạn $I = \lim \left(\sqrt[3]{8n^3 + 1} - \sqrt{4n^2 - n + 5}\right)$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \left(\sqrt[3]{8n^3 + 1} - \sqrt{4n^2 - n + 5}\right) \\
    &= \lim \left(\sqrt[3]{8n^3 + 1} - 2n\right) + \lim \left(2n - \sqrt{4n^2 - n + 5}\right) \\ 
    &= \lim \dfrac{8n^3 + 1 - 8n^3}{\sqrt[3]{(8n^3 + 1)^2} + 2n\sqrt[3]{8n^3 + 1} + 4n^2} + \lim \dfrac{4n^2 - (4n^2 - n + 5)}{2n + \sqrt{4n^2 - n + 5}} \\
    &= \lim \dfrac{1}{\sqrt[3]{(8n^3 + 1)^2} + 2n\sqrt[3]{8n^3 + 1} + 4n^2} + \lim \dfrac{n - 5}{2n + \sqrt{4n^2 - n + 5}} \\
    &= 0 + \lim \dfrac{1 - \frac{5}{n}}{2 + \sqrt{4 - \frac{1}{n} + \frac{5}{n^2}}} \\
    &= \dfrac{1}{2 + \sqrt{4}} = \dfrac{1}{4}
    \end{align*}
    }
    \end{bt}  
    \begin{bt}%[1K5?E-4]%[1K5?E-4]
        Tính giới hạn $I = \lim \dfrac{\sqrt{3n^2 + 1} - \sqrt{n - 1}}{n}$.
\loigiai{
    Ta có
    \begin{align*} 
    I &= \lim \dfrac{\sqrt{3n^2 + 1} - \sqrt{n - 1}}{n} \\
    &= \lim \dfrac{3n^2 + 1 - (n - 1)}{n \left(\sqrt{3n^2 + 1} + \sqrt{n - 1}\right)} \\ 
    &= \lim \dfrac{3n^2 - n + 2}{n \left(\sqrt{3n^2 + 1} + \sqrt{n - 1}\right)} \\
    &= \lim \dfrac{3 - \frac{1}{n} + \frac{2}{n^2}}{\sqrt{3 + \frac{1}{n^2}} + \sqrt{\frac{1}{n} - \frac{1}{n^2}}} \\
    &= \sqrt{3}
    \end{align*}
    }
    \end{bt}
