\begin{vd}%[2D1H4-1]
    Tiệm cận xiên của đồ thị hàm số $y=\dfrac{-x^2-3x+4}{x+2}$ là đường thẳng có phương trình?
    \choice
    {\True $y=-x-1$}
    {$y=x-1$}
    {$y=-x+1$}
    {$y=x+1$}
    \loigiai{
        Ta có 
        \begin{itemize}
            \item $a=\lim\limits_{x \to+\infty}\left(\dfrac{-x^2-3x+4}{x+2}: x\right)=\lim\limits_{x \to+\infty} \dfrac{-x^2-3x+4}{x^2+2x}=-1$.
            \item $b=\lim\limits_{x \to+\infty}\left[\dfrac{-x^2-3x+4}{x+2}-(-1) x\right]=\lim\limits_{x \to+\infty} \dfrac{-x+4}{x+2}=-1$.
        \end{itemize}
        Ngoài ra 
        \begin{itemize}
            \item $\lim\limits_{x \to-\infty}\left(\dfrac{-x^2-3x+4}{x+2}: x\right)=-1$.
            \item $\lim\limits_{x \to-\infty}\left[\dfrac{-x^2-3x+4}{x+2}-(-1) x\right]=-1$.
        \end{itemize}
        Do đó tiệm cận xiên của đồ thị hàm số $y=\dfrac{-x^2-3x+4}{x+2}$ là đường thẳng có phương trình $y=-x-1$.
    }
\end{vd}
