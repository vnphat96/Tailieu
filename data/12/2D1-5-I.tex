\begin{ex}%[2D1K5-1]%[Võ Thị Thùy Trang]%Câu 9.
	Cho hàm số $y=f(x)$ khác hàm hằng, xác định trên $\mathbb{R}$, có đạo hàm tại mọi điểm thuộc $\mathbb{R}$ và xác định trên $\mathbb{R}$. Xét $4$ mệnh đề sau đây
	\begin{itemize}
		\item $(I)\colon$ Số nghiệm của phương trình $f'(x)=0$ luôn bé hơn số nghiệm của phương trình $f(x)=0$.
		\item $(II)\colon$ Nếu $y=f(x)$ là hàm số chẵn thì $y=f'(x)$ là hàm số lẻ.
		\item $(III)\colon$ Tiếp tuyến của đồ thị hàm số $y=f(x)$ tại điểm có hoành độ $x_0$ có hệ số góc $k=f'(x_0)$.
		\item $(IV)\colon$ Nếu $f'(x_1)=f'(x_2)$ và $x_1\neq x_2$ thì tiếp tuyến của đồ thị hàm số $y=f(x)$ tại các điểm có hoành độ $x_1,x_2$ song song với nhau.
	\end{itemize}
	Số mệnh đề đúng là 
	\choice
	{$1$}
	{\True $2$}
	{$3$}
	{$4$}
	\loigiai{
		\begin{itemize}
			\item Mệnh đề Số nghiệm của phương trình $f'(x)=0$ luôn bé hơn số nghiệm của phương trình $f(x)=0$ \textbf{sai} vì lấy $y=\sin x,y=0\Leftrightarrow x=k\pi,k\in\mathbb{Z}$. Phương trình $y'=0\Leftrightarrow x=\dfrac{\pi}{2}+k\pi,k\in\mathbb{Z}$.
			\item Mệnh đề Nếu $y=f(x)$ là hàm số chẵn thì $y=f'(x)$ là hàm số lẻ đúng:\\
			Thật vậy, Do hàm số là hàm số chẵn nên ta có:\\
			$f(-x)=f(x)\Rightarrow-f'(-x)=f'(x)\Rightarrow f'(-x)=-f'(x)$. Mà hàm số $y=f(x)$ có đạo hàm tại mọi điểm thuộc $\mathbb{R}$ và xác định trên $\mathbb{R}$.\\
			Do đó ta có hàm số $y=f'(x)$ là hàm số lẻ.
			\item Dễ thấy mệnh đề Tiếp tuyến của đồ thị hàm số $y=f(x)$ tại điểm có hoành độ $x_0$ có hệ số góc $k=f'(x_0)$ đúng.
			\item Mệnh đề Nếu $f'(x_1)=f'(x_2)$ và $x_1\neq x_2$ thì tiếp tuyến của đồ thị hàm số $y=f(x)$ tại các điểm có hoành độ $x_1,x_2$ song song với nhau Sai: ví dụ như hàm $y=\sin x$; $y'=\cos x$; $y'=0\Leftrightarrow x=\dfrac{k\pi}{2}$, ($k\in\mathbb{Z}$). Tiếp tuyến của đồ thị hàm số tại các điểm này lại trùng nhau.
		\end{itemize}}
\end{ex}
