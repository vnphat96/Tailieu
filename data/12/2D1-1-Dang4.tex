\begin{vd}%Câu 4.%[Paul Hieu Nguyen]%[2D1K1-4]

	\loigiai{

\end{vd}
\begin{vd}%Câu 5.%[Paul Hieu Nguyen]%[2D1B1-4]
	Tìm tất cả các giá trị thực của tham số $a$ để phương trình $1-x^2=a\sqrt{1+x}$ có nghiệm $x\in[0;1]$.

\end{vd}
