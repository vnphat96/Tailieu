\setcounter{section}{1}
\section{LOGARIT}
\subsection{Kiến thức sách giáo khoa cần cần nắm}
\subsubsection{Định nghĩa}
Cho hai số dương $a$, $b$ với $a\neq 1$. Số $\alpha$ thỏa mãn đẳng thức $a^{\alpha}=b$ được gọi là lôgarit cơ số $a$ của $b$ và kí hiệu là $\log_ab$.\\
Ta viết $\alpha=\log_ab\Leftrightarrow a^{\alpha}=b$.
\subsubsection{Các tính chất}
Cho $a, b>0, a\neq 1$, ta có
\begin{itemize}
	\item $\log_aa=1,\log_a1=0$.
	\item $a^{\log_ab}=b,\log_a(a^{\alpha})=\alpha$.
\end{itemize}
\subsubsection{Lôgarit của một tích}
Cho 3 số dương $a$, $b_1$, $b_2$ với $a\neq 1$, ta có 
$$\log_a(b_1\cdot b_2)=\log_ab_1+\log_ab_2.$$
\subsubsection{Lôgarit của một thương}
Cho 3 số dương $a, b_1, b_2$ với $a\neq 1$, ta có
$$\log_a\dfrac{b_1}{b_2}=\log_ab_1-\log_ab_2.$$
\textbf{Đặc biệt}, với $a$, $b>0$, $a\neq 1$ ta có $\log_a\dfrac{1}{b}=-\log_ab$.
\subsubsection{Lôgarit của lũy thừa}
Cho $a, b>0, a\neq 1$, với mọi $\alpha$, ta có
$$\log_ab^{\alpha}=\alpha\log_ab.$$
Đặc biệt: $\log_a\sqrt[n]{b}=\dfrac{1}{n}\log_ab$.
\subsubsection{Công thức đổi cơ số}
Cho 3 số dương $a, b, c$ với $a\neq 1,c\neq 1$, ta có
$$\log_ab=\dfrac{\log_cb}{\log_ca}.$$
Đặc biệt: $\log_ac=\dfrac{1}{\log_ca}$ và $\log_{a^{\alpha}}b=\dfrac{1}{\alpha}\log_ab$ với $\alpha\neq 0$.\\
\subsubsection{Lôgarit thập phân và Lôgarit tự nhiên}
\begin{itemize}
	\item Lôgarit thập phân là lôgarit cơ số $10$. Viết $\log_{10}b=\log b=\lg b$.
	\item Lôgarit tự nhiên là lôgarit cơ số $\mathrm{e}$. Viết $\log_{\mathrm{e}}b=\ln b$.
\end{itemize}
\subsection{Phân loại và phương pháp giải bài tập}
\begin{dang}{Công thức lôgarit}
\end{dang}
\subsubsection{Các ví dụ}
\begin{vd}%Ví dụ 1.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2B3-1]
	Tính giá trị của biểu thức $P=\log_a\left(a\cdot\sqrt[3]{a\sqrt{a}}\right)$ với $0<a\neq 1$.
	\loigiai{
		Ta có $P=\log_a\left[a\cdot\left(a\cdot a^{\tfrac{1}{2}}\right)^{\tfrac{1}{3}}\right]=\log_a\left(a^{\tfrac{3}{2}}\right)=\dfrac{3}{2}\log_aa=\dfrac{3}{2}$.}
\end{vd}

\begin{vd}%Ví dụ 2.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2B3-1]
	Tính giá trị của biểu thức $P=22\log_212+3\log_25-\log_215-\log_2150$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
		P&=&2\log_212+3\log_25-\log_215-\log_2150\\
		&=&\log_2{12}^2+\log_25^3-\log_2(15\cdot 150)\\
		&=&\log_2\dfrac{{12}^2{\cdot 5}^3}{15\cdot 150}=\log_28=3.
		\end{eqnarray*}
		}
\end{vd}

\begin{vd}%Ví dụ 3.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2B3-1]
	Cho $a>0$, $a\neq 1$. Tính giá trị biểu thức $A=(\ln a+\log_a\mathrm{e})^2+\ln ^2a-\log_a^2\mathrm{e}$.
	\loigiai{
		Ta có $A=\ln ^2a+2\ln a\cdot\log_a\mathrm{e}+\log_a^2e+\ln ^2a-\log_a^2\mathrm{e}=2\ln ^2a+2\ln \mathrm{e}=2\ln ^2a+2$.}
\end{vd}

\begin{vd}%Ví dụ 4.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2B3-1]
	Cho các số dương $a$, $b$, $c$, $d$. Tính giá trị biểu thức $S=\ln \dfrac{a}{b}+\ln \dfrac{b}{c}+\ln \dfrac{c}{d}+\ln \dfrac{d}{a}$.
	\loigiai{
	Ta có $S=\ln \dfrac{a}{b}+\ln \dfrac{b}{c}+\ln \dfrac{c}{d}+\ln \dfrac{d}{a}=\ln \left(\dfrac{a}{b}\cdot\dfrac{b}{c}\cdot\dfrac{c}{d}\cdot\dfrac{d}{a}\right)=\ln 1=0$.}
\end{vd}

\begin{vd}%Ví dụ 5.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2B3-1]
	Tính giá trị của biểu thức $P=\log_{a^2}\left(a^{10}b^2\right)+\log_{\sqrt{a}}\left(\dfrac{a}{\sqrt{b}}\right)+\log_{\sqrt[3]{b}}b^{-2}$ (với $0<a\neq 1; 0<b\neq 1$).
	\loigiai{
		\textbf{Cách 1:} Sử dụng các quy tắc biến đổi logarit.
		Ta có
		\begin{eqnarray*}
		P&=&\log_{a^2}\left(a^{10}b^2\right)+\log_{\sqrt{a}}\left(\dfrac{a}{\sqrt{b}}\right)+\log_{\sqrt[3]{b}}b^{-2}\\
		&=&\dfrac{1}{2}\left[\log_aa^{10}+\log_ab^2\right]+2\left[\log_aa-\log_a\sqrt{b}\right]+3\cdot (-2)\log_bb\\
		&=&\dfrac{1}{2}\left[10+2\log_ab\right]+2\left[1-\dfrac{1}{2}\log_ab\right]-6=1.
		\end{eqnarray*}
		\textbf{Cách 2:} Ta thấy các đáp án đưa ra đều là các hằng số, như vậy ta dự đoán giá trị của $P$ không phụ thuộc vào giá trị của $a,b$.\\
		Khi đó, sử dụng máy tính cầm tay, ta tính giá trị của biểu thức khi $a=2$; $b=2$, ta được 
		$$P=\log_4\left(2^{10}\cdot 4\right)+\log_{\sqrt{2}}\left(\dfrac{2}{\sqrt{2}}\right)+\log_{\sqrt[3]{2}}2^{-2}=1.$$}
\end{vd}

\begin{vd}%Ví dụ 6.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2G3-2]
	Xét các số thực $a$, $b$ thỏa mãn $a>b>1$. Tìm giá trị nhỏ nhất $P_{\min}$ của biểu thức $P=\log_{\tfrac{a}{b}}^2(a^2)+3\log_b\left(\dfrac{a}{b}\right)$.
	\loigiai{
		Với điều kiện đề bài, ta có
		\begin{eqnarray*}
		P&=&\log_{\tfrac{a}{b}}^2(a^2)+3\log_b\left(\dfrac{a}{b}\right)=\left[2\log_{\tfrac{a}{b}}a\right]^2+3\log_b\left(\dfrac{a}{b}\right)\\
		&=&4\left[\log_{\tfrac{a}{b}}\left(\dfrac{a}{b}\cdot b\right)\right]^2+3\log_b\left(\dfrac{a}{b}\right)=4\left[1+\log_{\tfrac{a}{b}}b\right]^2+3\log_b\left(\dfrac{a}{b}\right)
		\end{eqnarray*}
		Đặt $t=\log_{\tfrac{a}{b}}b>0$ (vì $a>b>1$), ta có $$P=4(1+t)^2+\dfrac{3}{t}=4t^2+8t+\dfrac{3}{t}+4.$$
		Đặt $f(t)=4t^2+8t+\dfrac{3}{t}+4$, $f’(t)=8t+8-\dfrac{3}{t^2}=\dfrac{8t^3+8t^2-3}{t^2}=\dfrac{(2t-1)(4t^2+6t+3)}{t^2}$.\\
		Ta có $f’(t)=0\Leftrightarrow t=\dfrac{1}{2}$. 
	\begin{center}
			\begin{tikzpicture}

		Khảo sát hàm số, ta có $P_{\min} =f\left(\dfrac{1}{2}\right)=15$.}
\end{vd}

\begin{vd}%Ví dụ 7.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2K3-1]
	Tính giá trị của biểu thức $P=\ln \left(\tan1^{\circ}\right)+\ln \left(\tan2^{\circ}\right)+\ln \left(\tan3^{\circ}\right)+\cdots +\ln \left(\tan89^{\circ}\right)$.
	\loigiai{

		}

		&\Leftrightarrow& 2\log_2(x+y)=4+\log_2x+\log_2y\\
		&\Leftrightarrow&\log_2(x+y)=2+\dfrac{\log_2xy}{2}.
		\end{eqnarray*}
		 }
\end{ex}

\begin{ex}%Câu 39.%[Đỗ Đường Hiếu - ĐCHT THPT]%[2D2K3-2]
