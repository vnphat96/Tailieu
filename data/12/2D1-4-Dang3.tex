\begin{ex}%[2D1G4-1]%[Thầy Hải Toán]%Câu 8.

	\begin{center}
	\begin{tikzpicture}[>=stealth,scale=.8]
	\draw[->] (0,-3)--(0,5)node[right]{$y$};
	\foreach \y in {-2,-1,1,2,3,4}
	\draw[shift={(0,\y)}] (2pt,0)--(-2pt,0) node[left]{\scriptsize $\y$};
	\draw[->] (-3,0)--(5,0)node[below]{$x$};
	\foreach \x in {-2,-1,1,2,3,4}
	\draw[shift={(\x,0)}] (0,2pt)--(0,-2pt) node[below]{\scriptsize $\x$};
	\draw[dashed] (0,4)-|(1,0);
	\draw[samples=100,domain=0.4:3.5,smooth] plot (\x, {2.4*(\x)^3-15.74*(\x)^2+29.8*(\x)-12.53}) node[right]{$f(x)$};

	\end{tikzpicture}
	\end{center}

	{$5$}
	{$3$}
	{$6$}

	\begin{center}
	\begin{tikzpicture}[>=stealth,scale=1]
	\draw[->] (-2,0)--(4,0)node[below]{$x$};
	\foreach \x in {-1,1,2,3}

	\draw[->] (0,-3)--(0,3)node[right]{$y$};
	\foreach \y in {-2,-1,1,2}
	\draw[shift={(0,\y)}] (2pt,0)--(-2pt,0) node[right]{\scriptsize $\y$};
	\draw[samples=100,domain=0.85:2.4,smooth] plot (\x, {10.5*(\x)^3-52.27*(\x)^2+83.28*(\x)-41.47}) node[right]{$f(x)$};
	\fill[red] (0,0)circle(1.5pt);
	\end{tikzpicture}
	\end{center}
	\choice
	{$3$}
	{\True $5$}
	{$6$}
	{$4$}
	\loigiai{
	Nhìn vào đồ thị hàm số ta có:\\
	$f(x)=0\Leftrightarrow\hoac{&x=1\\&x=2.}$ \\
	$f(x)=1\Leftrightarrow\hoac{&x=a\in(1;2)\\&x=b\in(a;2)\\&x=c\in(2;3).}$ \\
	$g(x)=\dfrac{\left(x^2-3x+2\right)\sqrt{x-1}}{x\left[f^2(x)-f(x)\right]} =\dfrac{(x-1)(x-2)\sqrt{x-1}}{xf(x)[f(x)-1]}$.\\
	$=\dfrac{(x-1)(x-2)\sqrt{x-1}}{x(x-1)(x-2)^2(x-a)(x-b)(x-c)}$.\\
	$=\dfrac{\sqrt{x-1}}{x(x-2)(x-a)(x-b)(x-c)}$.\\
	Suy ra đồ thị hàm số $y=g(x)$ có $4$ tiệm cận đứng $x=2$, $x=a$, $x=b$, $x=c$ và tiệm cận ngang $y=0$.}
\end{ex}
\begin{ex}%[2D1G4-1]%[Thầy Hải Toán]%Câu 10.

	\draw[samples=100,smooth] (-0.9,3.2)..controls+(-90:0.5)and+(180:0.5)..(0,0)..controls+(10:0.2)and+(180:0.2)..(0.55,0.15)..controls+(0:0.2)and+(180:0)..(1,0)..controls+(0:0)and+(180:.2)..(1.6,-0.4)..controls+(0:0.2)and+(-110:.2)..(2,0)..controls+(70:0.5)and+(-110:0)..(2.5,3.2);
	\fill[red] (0,0)circle(1.5pt);
	\end{tikzpicture}
	\end{center}
	Hỏi đồ thị hàm số $y=\dfrac{f^2(x)\sqrt{x^2+x}}{\left[f^2(x)-2f(x)\right]\left(2x^5+x^4-10x^3-5x^2+8x+4\right)}$ có bao nhiêu tiệm cận đứng và ngang?
	\choice
	{$7$}
	{$6$}
	{$5$}
	{\True $4$}
	\loigiai{
	Dựa vào đồ thị ta có: $f(x)=a\cdot x^2(x-1)(x-2)$.\\
	Xét: $\left[f^2(x)-2f(x)\right]\left(2x^5+x^4-10x^3+8x+4\right)=0\Leftrightarrow\hoac{&f(x)=0\\&f(x)=2\\&2x^5+x^4-10x^3-5x^2+8x+4=0}$ \\
	$ \Leftrightarrow\hoac{&x=0\\&x=1\\&x=2\\&x=x_1(-1<x_1<0)\\&x=x_2(2<x_2<3)\\&x=\pm 2\\&x=\pm 1\\&x=-\dfrac{1}{2}} $ \\
