	{$24$}
	{$11$}
	{$16$}
	\loigiai{
		Gọi $A$ là tập hợp các bạn tham gia thi đấu bóng đá $\Rightarrow n(A)=16$.\\

		Ta có $n(A \cup B)=n(A)+n(B)-n(A \cap B)\Rightarrow n(A \cap B)=n(A)+n(B)-n(A \cup B)=16+11-24=3$.
	}
