			\item Mệnh đề ``$\forall x\in\mathbb{Z}\colon 2x>1$'' sai vì với $x=0$ thì $2x=0<1$.
			\item Mệnh đề ``$\exists x\in\mathbb{R}\colon x^2>x$'' đúng vì với $x=2$ thì $x^2=4$ nên $x^2>x$.
		\end{itemize}
	}
\end{ex}
